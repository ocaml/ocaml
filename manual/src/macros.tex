                        \addpenalty\@beginparpenalty

          \fi
    \global\@inlabeltrue
\fi

          \if@inlabel\global\@inlabelfalse \hskip -\parindent \box\@labels

 \hbox{\unhbox\@labels
       \hskip -\leftmargin
       \box\@tempboxa}\hfil\break
 \else

       \hskip -\leftmargin
       \hbox to\leftmargin {\makelabel{#1}\hfil}}
 \fi
 \ignorespaces}
